# Group分组

在`graphify-chart`，性能方面的表现是非常优异的。

```tsx preview
import { useEffect, useState, useRef } from 'react';
import {Stage,Circle} from 'graphify-chart';
import colorAlpha from 'color-alpha'
const radius = 25
const maxRadius = 50

function randomX() {
  return maxRadius + Math.random() * (750 - maxRadius * 2)
}

function randomY() {
  return maxRadius + Math.random() * (500 - maxRadius * 2)
}

function randomColor() {
  const r = Math.round(255 * Math.random())
  const g = Math.round(255 * Math.random())
  const b = Math.round(255 * Math.random())

  return `rgb(${r}, ${g}, ${b})`
}

const App = () => {
  const chartRef = useRef(null);
  const [list, setList] = useState([])

  useEffect(() => {
    const timer = setInterval(() => {
      createBubble()
    }, 100)

    return () => {
      clearInterval(timer)
    }
  }, [])

  function createBubble() {
    const radius = 25
    const maxRadius = 50

    const fillStyle = randomColor()
    const targetColor = colorAlpha(fillStyle, 0)

    const bubble = {
      key: Math.random().toString(),
      x: randomX(),
      y: randomY(),
      fillStyle: randomColor(),
      radius: radius,
      animationEnd: () => {
        setList(value => value.filter(item => item.key !== bubble.key))
      }
    }

    setList(value => value.concat(bubble))

    setTimeout(() => {
      bubble.x = randomX()
      bubble.y = randomY()
      bubble.radius = maxRadius
      bubble.fillStyle = targetColor

      setList(value => [...value])
    }, 10)
  }
  useEffect(() => {
   
    const stage = new Stage(chartRef.current);
       console.log(stage.destroy)
    list.forEach((option) => {
      const circle = new Circle(stage,option)
      stage.appendChild(circle)
    })
  
      return () => {
       stage?.destroy?.()
      }
  },[list])
  return ( <div ref={chartRef} style={{height:500}} ></div>)
}

export default App
```
