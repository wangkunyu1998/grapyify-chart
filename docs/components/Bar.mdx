

# 柱状图

```tsx
import {useEffect,useState,useRef} from 'react';
import {Stage,BarChart} from 'graphify-chart';
const options = 
      {
      barColor: ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEEAD'],
      margin: 60,
      data: [
    { label: '1月', value: 65 },
    { label: '2月', value: 80 },
    { label: '3月', value: 45 },
    { label: '4月', value: 95 },
    { label: '5月', value: 70 }
  ],
    }
function App() {
  const chartRef = useRef(null);
  useEffect(() => {
    const stage = new Stage(chartRef.current);
    new BarChart(stage,options);
  },[])
  return (<div id="rect" ref={chartRef} style={{height:500}} ></div>)
}
export default App;
```

## 属性支持

| 属性 | 说明  | 类型 | 默认值 |
| :-- | :---: | :--- | :---: |
| margin | 图表与容器的边距 | ?number | 30 |
| barColor | 柱状颜色 | ?string[] | ['#4A90E2', '#7ED321', '#F5A623'] |
| animation | 是否开启动画 | ?boolean | true |
| animationDuration | 动画持续时间（ms） | ?number | 1000 |
| labelStyle | 坐标轴字体样式 | ?\{fontSize:number,fontFamily:string,color:string} | \{fontSize:12,fontFamily:默认字体,color:\#333} |
| axisStyle | 坐标轴样式 | ?\{color:string,lineWidth:number} | \{color:\#666,lineWidth:1 } |
| data | 数据 | \{ label:string｜number, value: number }[] | 无 |
