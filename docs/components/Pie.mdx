

# 饼图

## 示例
```tsx preview
import {useEffect,useState,useRef} from 'react';
import {Stage,PieChart} from 'graphify-chart';
  const options= {
      radius:150,
      innerRadius: 0,
      colors: ['#5470c6', '#91cc75', '#fac858', '#ee6666', '#73c0de'],
      data:  [
        { label: '食品', value: 4200 },
        { label: '住房', value: 3100 },
        { label: '交通', value: 2400 },
        { label: '娱乐', value: 1500 },
        { label: '其他', value: 800 }
      ]
  }
function App() {
  const chartRef = useRef(null);
  useEffect(() => {
    const stage = new Stage(chartRef.current);
     new PieChart(stage,options )
  },[])
  return (<div id="rect" ref={chartRef} style={{height:300,width:'100%'}} ></div>)
}
export default App;
```
## 属性支持

| 属性 | 说明  | 类型 | 默认值 |
| :-- | :---: | :--- | :---: |
| margin | 图表与容器的边距 | ?number | 30 |
| radius | 饼图半径 | ?number | 150 |
| innerRadius | 内圆半径 | ?number | 0（表示无内圆环） |
| colors | 饼图颜色 | ?string[] | ['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEEAD'] |
| labelStyle | 坐标轴字体样式 | ?\{fontSize:number,fontFamily:string,color:string} | \{fontSize:12,fontFamily:默认字体,color:\#333} |
| data | 数据 | \{ label:string｜number, value: number }[] | 无 |
